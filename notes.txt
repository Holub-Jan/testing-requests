All data:
organizations as or
------------
org_id PK int
name varchar(200) UNIQUE
active int

repositories as re
------
repo_id PK int
org_id int FK >- or.org_id
name varchar(200) UNIQUE

teams as tm
------
team_id PK int
org_id int FK >- or.org_id
name string

team_repos as tr
----
team-repos-id PK int
team_id int FK >- tm.team_id
repo_id int FK >- re.repo_id
name string
role string

users as u
----
user_id PK int
team_id int FK >- tm.team_id
name UNIQUE string

keys as k?
